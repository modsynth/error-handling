# Error Handling

> Error Boundary error handling module

## Installation
```bash
npm install @modsynth/error-handling
```

## Usage
```tsx
import { ErrorBoundary } from '@modsynth/error-handling';

<ErrorBoundary fallback={<div>Error occurred</div>}>
  <App />
</ErrorBoundary>
```
